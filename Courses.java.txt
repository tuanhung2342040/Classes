class Courses {
    private String name;
    private int grade;
    Courses(){
        grade = 78;
        name="";
    }
    Courses(String name, int grade){
        this.name=name;
        this.grade=grade;
    }

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    public int getGrade() {
        return grade;
    }

    public void setGrade(int grade) {
        this.grade = grade;
    }
    public String toString(){
        return"Course:" + name;
    }
}
